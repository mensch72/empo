# Optional dependencies for hierarchical RL with LLMs and Minecraft
# These are large packages and are not included in the base Docker image by default.
# Install with: pip install -r requirements-hierarchical.txt

# Ollama - Python client for local LLM inference via Ollama server
# Note: This is just the client library. The Ollama server runs separately,
# either as a system service or in its own Docker container.
# See docker-compose.yml for optional Ollama service configuration.
# See: https://github.com/ollama/ollama-python
ollama>=0.1.0

# Pillow - Image processing library (needed for vision LLM integration)
Pillow>=10.0.0

# MineRL - Reinforcement Learning in Minecraft
# Note: MineRL bundles Minecraft/Malmo internally - no separate server needed.
# System requirements (handled in Dockerfile when HIERARCHICAL_MODE=true):
#   - Java JDK 8 (strict requirement)
#   - xvfb for headless rendering in Docker
# See: https://minerl.readthedocs.io/en/latest/tutorials/index.html
minerl>=1.0.0
